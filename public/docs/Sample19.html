                MgmtApi mgmtApi = new MgmtApi();
                mgmtApi.setBasePath(credentials.basePath);

                if (!StringUtils.isEmpty(callbackUrl)) {
                    FileOutputStream fileOutputStream = new FileOutputStream(USER_INFO_FILE);
